session_cache AS (

       p.value - c.value session_cache_misses,

   AND n1.name = 'session cursor cache hits'
   AND n2.inst_id = p.inst_id
   AND n2.statistic# = p.statistic#
   AND n2.name = 'parse count (total)'

)
