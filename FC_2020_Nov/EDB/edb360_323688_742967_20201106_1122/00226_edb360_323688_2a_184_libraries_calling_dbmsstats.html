),
include_nearby_lines AS (
SELECT /*+  MATERIALIZE NO_MERGE  */ /* 2a.184 */ DISTINCT s.*
  FROM lines_with_api l,
       dba_source s
 WHERE 'Y' = 'Y'
   AND s.owner = l.owner
   AND s.name = l.name
   AND s.type = l.type
   AND s.line BETWEEN l.line - 8 AND l.line + 8
