       con_id,
       sql_id,
       dbid,
       ROW_NUMBER () OVER (ORDER BY COUNT(*) DESC) rn,
       COUNT(*) samples
  FROM dba_hist_active_sess_history h
 WHERE instance_number = 2 AND TO_CHAR(sample_time, 'D') BETWEEN '2' AND '6' AND TO_CHAR(sample_time, 'HH24') BETWEEN '0730' AND '1930'
   AND sql_id IS NOT NULL
   AND snap_id BETWEEN 3473 AND 4962
   AND dbid = 2109323688
 GROUP BY
       con_id,
       sql_id,
       dbid
),
total AS (
SELECT /*+  MATERIALIZE NO_MERGE  */ /* 6c.812 */ SUM(samples) samples FROM hist
)
SELECT DISTINCT
       h.sql_id,
       h.samples,
       ROUND(100 * h.samples / t.samples, 1) percent,
       DBMS_LOB.SUBSTR(s.sql_text, 1000) sql_text
  FROM hist h,
       total t,
       dba_hist_sqltext s
 WHERE h.samples >= t.samples / 1000 AND rn <= 14

  FROM hist h,
       total t
